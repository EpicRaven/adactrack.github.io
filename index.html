Google Sheets Integration
Follow the steps below to enable real data storage and retrieval via Google Apps Script.
1
Create a Google Sheet
Go to sheets.google.com, create a new sheet, and add these headers in Row 1:
Timestamp | UserID | FullName | MatricNo | RegNo | Email | Semester | Session | Courses_JSON | GPA | GradeClass | Status | SubmittedBy
2
Paste the Apps Script
Extensions → Apps Script → replace all code with below → Save → Deploy → Web App (Anyone)
const SH="Sheet1"; function doPost(e){const s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SH),d=JSON.parse(e.postData.contents);s.appendRow([new Date().toISOString(),d.userID,d.fullName,d.matricNo,d.regNo,d.email,d.semester,d.session,JSON.stringify(d.courses),d.gpa,d.gradeClass,"Pending",d.submittedBy]);return out({status:"ok"})} function doGet(e){const s=SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SH),rows=s.getDataRange().getValues(),uid=(e.parameter.uid||"").trim().toUpperCase(),mat=(e.parameter.mat||"").trim().toUpperCase(),sem=e.parameter.sem||"",all=e.parameter.all==="1";const match=rows.slice(1).filter(r=>String(r[1]).toUpperCase()===uid&&String(r[3]).toUpperCase()===mat&&(all||String(r[6])===sem));if(!match.length)return out({status:"not_found"});return out({status:"ok",records:match.map(r=>({userID:r[1],fullName:r[2],matricNo:r[3],regNo:r[4],email:r[5],semester:r[6],session:r[7],courses:JSON.parse(r[8]||"[]"),gpa:r[9],gradeClass:r[10],status:r[11]}))})} function out(d){return ContentService.createTextOutput(JSON.stringify(d)).setMimeType(ContentService.MimeType.JSON)}
3
Paste your Web App URL
